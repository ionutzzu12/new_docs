<hint type="info">
  **Note**: When the *Create table* checkbox is not selected, the table format (columns) that you will create must still adhere to the following format.&#x20;
</hint>

```mysql
#CREATE TABLE IF NOT EXISTS `TABLE_NAME`
(
	id          BIGINT AUTO_INCREMENT NOT NULL,
	record_uuid CHAR(36)              NOT NULL,
	arrived_at  DATETIME              NOT NULL,
	device_id   VARCHAR(64)           NULL,
	register_id VARCHAR(64)           NULL,
	tag_name    VARCHAR(64)           NULL,
	datatype    VARCHAR(32)           NULL,
	value       TEXT                  NULL,
	success     TINYINT               NULL,
	PRIMARY KEY (id)
);
```
